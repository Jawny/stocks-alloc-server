// @ts-check
/** Source file merging all external routes.
 * @module app
 * @requires express
 * @requires module:routes/yahooApi
 */
import express from "express";
import { yahooApiRoutes } from "@routes";

/**
 * Express app to mount external routes.
 * @type {object}
 * @const
 * @namespace app
 */
const app = express();

/**
 * Route serving stock summary
 * @name app initialization
 * @function
 * @memberof module:app
 */
app.use(express.json());
app.use(express.urlencoded({ extended: false }));
app.use("/api", yahooApiRoutes);

app.listen(3000, () => {
  console.log("Example app listening on port 3000!");
});
